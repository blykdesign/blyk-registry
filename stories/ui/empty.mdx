# Empty

Placeholder for empty states (no results, empty lists). Compound component with header, title, description, and content slots.

## Installation

```bash
npx shadcn@latest add "https://registry.blyk.design/r/base/empty.json"
```

## Usage

```tsx
import {
  Empty,
  EmptyHeader,
  EmptyTitle,
  EmptyDescription,
  EmptyContent,
  EmptyMedia,
} from "@/components/ui/empty"
```

```tsx
<Empty>
  <EmptyHeader>
    <EmptyMedia variant="icon">
      <Inbox className="size-6" />
    </EmptyMedia>
  </EmptyHeader>
  <EmptyContent>
    <EmptyTitle>No results found</EmptyTitle>
    <EmptyDescription>Try adjusting your search or filters.</EmptyDescription>
  </EmptyContent>
</Empty>
```
